Direct communication with Meta Graph API

Implementation Detail: sendMessageFacebook
async function sendMessageFacebook(to, text, credentials) {
  const { pageAccessToken } = credentials;
  
  // 1. Prepare Payload
  const payload = {
    recipient: { id: to },
    message: { text: text },
    messaging_type: "RESPONSE"
  };

  // 2. Dispatch to Meta
  const response = await fetch(
    `https://graph.facebook.com/v20.0/me/messages?access_token=${pageAccessToken}`,
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    }
  );

  return await response.json();
}
